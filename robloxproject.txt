local function checkGamePass(player, gamePassId, nameOfValue)
	coroutine.resume(coroutine.create(function()
		local ValueObj = player:FindFirstChild(nameOfValue)
		if not ValueObj then
			warn("Player does not have "..nameOfValue.." gamepass value object! ID: "..tostring(gamePassId))
			return
		end
		if player and player:IsA("Player") and ValueObj then
			-- Use pcall to handle potential errors
			local success, hasGamePass = pcall(function()
				return player:HasPass(gamePassId)
			end)

			if success then
				if hasGamePass then
					ValueObj.Value = 1
				else
					ValueObj.Value = 0
				end
			else
				warn("Error checking game pass for player: " .. player.Name)
			end
		end
	end))
end

function CheckAllGamePasses(player)
	checkGamePass(player, 870007852, "VIPPunchingBagPass")
end

-- NEW FUNCTION: Print all players in datastore
local function printAllPlayersInDatastore()
	print("=== SCANNING DATASTORE FOR ALL PLAYERS ===")

	local uniqueUserIds = {}
	local totalKeysFound = 0

	-- Iterate through all stats to find unique user IDs
	for _, statName in ipairs(allStats) do
		print("Scanning for stat: " .. statName)

		local success, pages = pcall(function()
			return CoinsStore:ListKeysAsync(statName, 100) -- Get up to 100 keys at a time
		end)

		if success then
			while true do
				local keys = pages:GetCurrentPage()

				for _, keyInfo in ipairs(keys) do
					totalKeysFound = totalKeysFound + 1
					local key = keyInfo.KeyName

					-- Extract user ID from key (assuming format is "StatNameUserID")
					local userId = string.gsub(key, statName, "")

					-- Check if it's a valid number
					if tonumber(userId) then
						uniqueUserIds[userId] = true
					end
				end

				if pages.IsFinished then
					break
				else
					pages:AdvanceToNextPageAsync()
				end
			end
		else
			warn("Failed to list keys for stat: " .. statName)
		end
	end

	-- Convert to array and print results with usernames
	local playerCount = 0
	local userIdList = {}

	for userId, _ in pairs(uniqueUserIds) do
		playerCount = playerCount + 1
		table.insert(userIdList, userId)
	end

	-- Sort the user IDs for better readability
	table.sort(userIdList, function(a, b) return tonumber(a) < tonumber(b) end)

	print("=== DATASTORE SCAN RESULTS ===")
	print("Total keys found: " .. totalKeysFound)
	print("Unique players found: " .. playerCount)
	print("Players:")

	-- Get usernames for each user ID
	local Players = game:GetService("Players")
	local UserService = game:GetService("UserService")

	for i = 1, #userIdList, 50 do -- Process in batches of 50 (API limit)
		local batch = {}
		for j = i, math.min(i + 49, #userIdList) do
			table.insert(batch, tonumber(userIdList[j]))
		end

		local success, userInfos = pcall(function()
			return UserService:GetUserInfosByUserIdsAsync(batch)
		end)

		if success and userInfos then
			for _, userInfo in ipairs(userInfos) do
				print(userInfo.Username .. " (Nickname: " .. userInfo.DisplayName .. ") (ID: " .. userInfo.Id .. ")")
			end
		else
			warn("Failed to fetch user info for batch starting at index " .. i)
		end

		task.wait(0.1) -- avoid hitting rate limits
	end


	print("=== END OF DATASTORE SCAN ===")
end

-- Function to get player info from datastore by User ID
local function getPlayerDataFromDatastore(userId)
	print("=== PLAYER DATA FOR USER ID: " .. userId .. " ===")

	for _, statName in ipairs(allStats) do
		local success, value = pcall(function()
			return CoinsStore:GetAsync(statName .. userId)
		end)

		if success and value ~= nil then
			print(statName .. ": " .. tostring(value))
		else
			print(statName .. ": No data or error")
		end
	end

	print("=== END OF PLAYER DATA ===")
end


-- Function to create a new stat value for a player
local function createStatValue(player, statName)
	local value
	-- Use NumberValue for Max PPE, IntValue for everything else
	if statName == "Max PPE" then
		value = Instance.new("NumberValue")
	else
		value = Instance.new("IntValue")
	end

	value.Name = statName
	if table.find(nonLeaderboardStats, statName) then
		value.Parent = player
	else
		value.Parent = player.leaderstats
	end
	return value
end

-- Function to retrieve a stat value for a player from the server
local function getStatValueFromServer(player, statName)
	local userId = player.UserId
	local value = CoinsStore:GetAsync(statName..userId)
	return value or 0
end

-- Function to save a stat value for a player to the server
local function saveStatValueToServer(player, statName, value)
	local userId = player.UserId
	CoinsStore:SetAsync(statName..userId, value)
end

-- Function to set the value of a stat for a player
local function setStatValue(player, statName, value)
	if player:FindFirstChild(statName) then
		player[statName].Value = value
	elseif player.leaderstats:FindFirstChild(statName) then
		player.leaderstats[statName].Value = value
	else
		createStatValue(player, statName).Value = value
	end
	saveStatValueToServer(player, statName, value)
end

-- Function to initialize the stats for a new player
local function initStats(player)
	for _, stat in ipairs(allStats) do
		local value = getStatValueFromServer(player, stat)
		-- Set minimum values for Strength and MaxLevel
		if stat == "Strength" and value < 1 then
			value = 1
		elseif stat == "MaxLevel" and value < 1 then
			value = 1
		end
		setStatValue(player, stat, value)
	end
	-- ADD SOMETHING INTO PLAYER THAT SAYS WHEN ALL STATS LOADED
	local inst = Instance.new("StringValue")
	inst.Name = "AllStatsLoaded"
	inst.Parent = player
end

		AimOfTimesJoined.Value = TimesJoined.Value + 5
		WaitForAimMet.Value = 0
	end

	if WaitForAimMet.Value == 0 and TimesJoined.Value >= AimOfTimesJoined.Value then
		print(AimOfTimesJoined.Value, TimesJoined.Value)
		PrimaryCurrency.Value = math.floor(PrimaryCurrency.Value * 1.25)
		-- Set up for the next day
		NewDay.Value = currentTimestamp + 19 * 60 * 60 -- 19 hours because why not
		WaitForAimMet.Value = 1
	end

function getlevelboundaries(folder)
	local ListOfBoundaries = {}
	local boundaries = folder:FindFirstChild("RockSpawnBoundaries")
	if boundaries then
		for _, boundaryFolder in pairs(boundaries:GetChildren()) do
			local boundaryMin = nil
			local boundaryMax = nil
			if boundaryFolder:IsA("Folder") then
				for x, part in pairs(boundaryFolder:GetChildren()) do
					part.Transparency = 1
					part.Locked = true
					part.CanCollide = false
					part.CanTouch = false
					part.Anchored = true
					if x == 1 then
						boundaryMin = part
					else
						boundaryMax = part
					end
				end
				local minBound = boundaryMin.Position + boundaryMin.Size / 2
				local maxBound = boundaryMax.Position - boundaryMax.Size / 2


				if minBound.X > maxBound.X then
					local temp = minBound
					minBound = Vector3.new(maxBound.X, minBound.Y, minBound.Z)
					maxBound = Vector3.new(temp.X, maxBound.Y, maxBound.Z)
				end
				if minBound.Y > maxBound.Y then
					local temp = minBound
					minBound = Vector3.new(minBound.X, maxBound.Y, minBound.Z)
					maxBound = Vector3.new(maxBound.X, temp.Y, maxBound.Z)
				end
				if minBound.Z > maxBound.Z then
					local temp = minBound
					minBound = Vector3.new(minBound.X, minBound.Y, maxBound.Z)
					maxBound = Vector3.new(maxBound.X, maxBound.Y, temp.Z)
				end
				table.insert(ListOfBoundaries,{minBound,maxBound})

			end
		end
	end
	return(ListOfBoundaries)
end


function insertRocks(level)
	local GenerationRate = 20
	local timesExecuted = 0
	coroutine.resume(coroutine.create(function()
		local ListOfBoundaries = getlevelboundaries(level)
		local continuousRefresh = false	
		if timesExecuted >= 1 then
			continuousRefresh = true
		end
		timesExecuted = timesExecuted + 1
		local RockFolder = level:FindFirstChild("Rocks")
		if not RockFolder then
			RockFolder = Instance.new("Folder")
			RockFolder.Parent = level
			RockFolder.Name = "Rocks"
		end

		local minimumDistance = level.DistanceBetweenRocks.Value
		local MaxRocks = math.huge --how many rocks you want to be spawned

		local currentAmountofRocks = 0
		local attempts = 0
		while (currentAmountofRocks <= MaxRocks and attempts < 200) or (continuousRefresh and attempts < 200) do


			attempts = attempts + 1
			local children = RockFolder:GetChildren()
			currentAmountofRocks = #children
			local areas = #ListOfBoundaries

			--randomly selects area
			local targetArea = ListOfBoundaries[math.random(1,areas)]

			local minBound = targetArea[1]
			local maxBound = targetArea[2]

			local randomX = math.random(minBound.X, maxBound.X)
			local randomY = math.random(minBound.Y, maxBound.Y)
			local randomZ = math.random(minBound.Z, maxBound.Z)

			local spawnPosition = Vector3.new(randomX, randomY, randomZ)

			local function calculateDistance(pos1, pos2)
				return (pos1 - pos2).magnitude
			end

			-- Iterate through each object in the folder
			local allow = true
			for _, object in pairs(RockFolder:GetChildren()) do
				if object:IsA("Model") then
					local objectPosition = object.WorldPivot.Position
					local distance = calculateDistance(spawnPosition, objectPosition)

					if distance < minimumDistance then
						allow = false
					else
						if allow == true then
							allow = true
						end
					end
				end
			end

			if allow then
				local AvailableRocksFolder = game.ServerStorage.Rocks:FindFirstChild(tostring(level.Name:sub(1,1)))

				local objects = {}
				local objectsWeighting = {}

				-- Function to add rocks from a folder to the objects list
				local function addObjectsFromFolder(folder)
					local function search(size)
						local found = false
						for _, rock in pairs(folder:GetChildren()) do
							if rock:IsA("Model") then
								if rock:FindFirstChild("Size") and rock:FindFirstChild("Weighting") then
									if rock:FindFirstChild("Size").Value == size then
										table.insert(objects, rock)
										table.insert(objectsWeighting, rock.Weighting.Value)
										found = true
									end
								end
							end
						end

						if found then
							size = size + 1
							search(size)
						end
					end
					search(1)
				end

				-- Add rocks from the level-specific folder
				addObjectsFromFolder(AvailableRocksFolder)

				-- Add collection models from within the same level folder
				local CollectionFolder = AvailableRocksFolder:FindFirstChild("Collection")
				if CollectionFolder then
					for _, collectionModel in pairs(CollectionFolder:GetChildren()) do
						if collectionModel:IsA("Model") and collectionModel:FindFirstChild("Weighting") then
							table.insert(objects, collectionModel)
							table.insert(objectsWeighting, collectionModel.Weighting.Value)
						end
					end
				end

				local billboardgui = AvailableRocksFolder:FindFirstChildWhichIsA("BillboardGui")



				local totalWeight = 0
				for _, obj in pairs(objectsWeighting) do
					totalWeight = totalWeight + obj
				end

				-- Function to choose a random object based on weights
				local function chooseRandomObject()
					local randomNumber = math.random(1, totalWeight)
					local currentWeight = 0
					for i, obj in pairs(objects) do
						currentWeight = currentWeight + objectsWeighting[i]
						if randomNumber <= currentWeight then
							-- Only add billboard gui if it's a rock (not a collection)
							if obj.Parent == AvailableRocksFolder and billboardgui then
								if not obj:FindFirstChildWhichIsA("BillboardGui") then
									local clone = billboardgui:Clone()
									clone.Parent = obj
									clone:FindFirstChildWhichIsA("Frame").Visible = false
								end
							end
							return obj
						end
					end
				end

				local chosenObject = chooseRandomObject()

				local part = chosenObject:Clone()
				if part:FindFirstChildWhichIsA("BillboardGui") then
					local gui = part:FindFirstChildWhichIsA("BillboardGui")


					if gui:FindFirstChildWhichIsA("Frame") then
						gui:FindFirstChildWhichIsA("Frame").Visible = false
					end
				end

local function initializePlayerMoneyData(player)
	ServerModule.MoneyRateData[player.UserId] = {
		player = player,
		moneyHistory = {}, -- This will store {timestamp, money} pairs
		currentRate = 0,
		joinTime = tick() -- Track when the player joined
	}
end

local function getPlayerStrength(player)
	local leaderstats = player:FindFirstChild("leaderstats")
	if leaderstats then
		local strength = leaderstats:FindFirstChild("Strength")
		if strength then
			return strength.Value
		end
	end
	return 1
end

local function addMoneyToHistory(player, money)
	local data = ServerModule.MoneyRateData[player.UserId]
	if data then
		-- Add new money entry with current timestamp
		table.insert(data.moneyHistory, {tick(), money})
	end
end

local function cleanOldMoneyHistory(data)
	local currentTime = tick()
	local cutoffTime = currentTime - 60 -- 60 seconds ago

	-- Remove entries older than 60 seconds
	local i = 1
	while i <= #data.moneyHistory do
		if data.moneyHistory[i][1] < cutoffTime then
			table.remove(data.moneyHistory, i)
		else
			i = i + 1
		end
	end
end

local function calculateMoneyInLast60Seconds(data)
	cleanOldMoneyHistory(data)

	local totalMoney = 0
	for _, entry in pairs(data.moneyHistory) do
		totalMoney = totalMoney + entry[2] -- entry[2] is the money amount
	end

	return totalMoney
end

local function getEffectiveTimeWindow(data)
	local currentTime = tick()
	local timeInServer = currentTime - data.joinTime
	-- Use the smaller of: time in server or 60 seconds
	return math.min(timeInServer, 60)
end

local function checkToUpdateMaxPPE(player, newVal)
	newVal = tonumber(string.format("%.2f",newVal))
	local success, result = pcall(function()
		local maxPPE = player.leaderstats:FindFirstChild("Max PPE")
		local oldVal = maxPPE.Value
		if newVal > oldVal then
			maxPPE.Value = newVal
		end
	end)

end

local function calculateMoneyRates()
	for userId, data in pairs(ServerModule.MoneyRateData) do
		local player = data.player
		if player and player.Parent then -- Make sure player still exists
			local strength = getPlayerStrength(player)
			local moneyInTimeWindow = calculateMoneyInLast60Seconds(data)
			local effectiveTimeWindow = getEffectiveTimeWindow(data)

			-- Only calculate rate if player has been in server for more than 0 seconds
			if effectiveTimeWindow > 0 then
				-- Calculate rate per second, then multiply by 60 to get per-minute rate
				local ratePerSecond = moneyInTimeWindow / effectiveTimeWindow
				data.currentRate = (ratePerSecond * 60) / strength
			else
				data.currentRate = 0
			end

			checkToUpdateMaxPPE(player, data.currentRate)
			--print(player.Name .. " - PPE: " .. string.format("%.2f", data.currentRate) .. 
			--	" (Money in time window: " .. tostring(moneyInTimeWindow) .. 
			--	", Effective time: " .. string.format("%.1f", effectiveTimeWindow) ..
			--	", Strength: " .. tostring(strength) .. ")")
		end
	end
end

function module.evolvePet(player, pet)
	local petFolder = player.Pets
	local petName = pet.Name

	local function checkForPet(pet)
		for _, item in pairs(petFolder:GetChildren()) do
			if item == pet then
				return true
			end
		end
		return false
	end

	local function returnEvolutionOf(pet)
		if pet:FindFirstChild("NumberOfEvolutions") then
			return pet:FindFirstChild("NumberOfEvolutions").Value
		else
			error("Couldn't find pet number of evolutions")
			return
		end
	end

	if checkForPet(pet) then
		local evo = returnEvolutionOf(pet)
		local sameEvoPets = {}

		for _, item in pairs(petFolder:GetChildren()) do
			if item.Name == petName and returnEvolutionOf(item) == evo then
				table.insert(sameEvoPets, item)
			end
		end

		if #sameEvoPets >= 3 then
			-- Remove the three pets
			for i = 1, 3 do
				sameEvoPets[i]:Destroy()
			end

			-- Add the evolved pet
			local ServerPetStorage = game.ServerStorage.Pets
			local evolvedPet = ServerPetStorage:FindFirstChild(petName):Clone()
			if evolvedPet then
				local evolutionValue = evolvedPet:FindFirstChild("NumberOfEvolutions")
				if evolutionValue then
					evolutionValue.Value = evo + 1
				else
					evolutionValue = Instance.new("IntValue")
					evolutionValue.Name = "NumberOfEvolutions"
					evolutionValue.Value = evo + 1
					evolutionValue.Parent = evolvedPet
				end
				evolvedPet.Parent = petFolder
			else
				error("Failed to find pet in Server Storage: " .. petName)
			end
		end
	end
end

	-- Connect remote event
	connections[#connections + 1] = antiExploitRemote.OnServerEvent:Connect(onRemoteRequest)

	connections[#connections + 1] = Players.PlayerRemoving:Connect(onPlayerRemoving)
	connections[#connections + 1] = RunService.Heartbeat:Connect(checkPlayerSpeeds)

	for _, player in pairs(Players:GetPlayers()) do
		onPlayerAdded(player)
	end

	print("AntiExploit system activated!")
	return true
end

local ppeHistory = {} -- [userId] = {values = {}, timestamps = {}, isDecreasing = false, decreaseStartTime = nil}

-- Function to get color based on PPE value and max PPE
local function getPPEColor(ppeValue, maxPPE)
	local hue

	-- Calculate the threshold (60% of max PPE)
	local greenThreshold = maxPPE * 0.6

	if ppeValue <= 0 then
		-- Completely red (hue = 0)
		hue = 0
	elseif ppeValue >= greenThreshold then
		-- Completely green (hue = 0.333)
		hue = 0.333
	else
		-- Interpolate between red (0) and green (0.333) from 0 to 60% of max PPE
		local normalizedValue = ppeValue / greenThreshold -- Normalize to 0-1 range
		hue = normalizedValue * 0.333 -- Map to hue range 0 to 0.333
	end

	-- Use Roblox's built-in HSV to RGB conversion
	return Color3.fromHSV(hue, 1, 1)
end

-- Function to update PPE trend tracking
local function updatePPETrend(player, ppeValue)
	local userId = player.UserId
	local currentTime = tick()

	-- Initialize history if it doesn't exist
	if not ppeHistory[userId] then
		ppeHistory[userId] = {
			values = {},
			timestamps = {},
			isDecreasing = false,
			decreaseStartTime = nil,
			lastTweenState = nil
		}
	end

	local history = ppeHistory[userId]

	-- Add current value to history
	table.insert(history.values, ppeValue)
	table.insert(history.timestamps, currentTime)

	-- Keep only last 10 values for trend analysis
	if #history.values > 10 then
		table.remove(history.values, 1)
		table.remove(history.timestamps, 1)
	end

	-- Analyze trend if we have enough data points
	if #history.values >= 3 then
		local isCurrentlyDecreasing = true
		local isCurrentlyIncreasing = true

		-- Check if trend is decreasing or increasing
		for i = 2, #history.values do
			if history.values[i] > history.values[i-1] then
				isCurrentlyDecreasing = false
			elseif history.values[i] < history.values[i-1] then
				isCurrentlyIncreasing = false
			end
		end

		-- Update decrease tracking
		if isCurrentlyDecreasing and not history.isDecreasing then
			-- Just started decreasing
			history.isDecreasing = true
			history.decreaseStartTime = currentTime
		elseif not isCurrentlyDecreasing then
			-- No longer decreasing
			history.isDecreasing = false
			history.decreaseStartTime = nil
		end

		-- Return trend information
		return {
			isDecreasing = history.isDecreasing,
			decreaseDuration = history.decreaseStartTime and (currentTime - history.decreaseStartTime) or 0,
			isIncreasing = isCurrentlyIncreasing and not isCurrentlyDecreasing
		}
	end

	return {isDecreasing = false, decreaseDuration = 0, isIncreasing = false}
end

-- Function to tween frame transparency
local function tweenFrameTransparency(frame, targetTransparency, userId, seconds)
	if not frame then return end

	local history = ppeHistory[userId]
	if not history then return end

	-- Don't start new tween if already tweening to same state
	if history.lastTweenState == targetTransparency then return end
	history.lastTweenState = targetTransparency
	
	if not seconds then
		seconds = 0.5
	end
	local tweenInfo = TweenInfo.new(
		seconds, -- Duration
		Enum.EasingStyle.Quad,
		Enum.EasingDirection.Out
	)

	local tween = TweenService:Create(frame, tweenInfo, {
		GroupTransparency = targetTransparency,
	})

	tween:Play()
end

-- Function to update PPE display for a player
local function updatePPEDisplay(player, ppeValue)
	-- Format the PPE value to 2 decimal places
	local formattedPPE = string.format("%.2f", ppeValue or 0)

	-- Check if the player has a character
	if player and player.Character then
		local humanoidRootPart = player.Character:FindFirstChild("HumanoidRootPart")
		if humanoidRootPart then
			local nameDisplay = humanoidRootPart:FindFirstChild("CustomNameDisplay")
			if nameDisplay then
				local frame = nameDisplay:FindFirstChild("Frame")
				if frame then
					local innerFrame = frame:FindFirstChild("Frame")
					if innerFrame then
						-- Try to find existing PPE label
						local ppeLabel = innerFrame.PPE
						ppeLabel.Text = formattedPPE

						-- Get Max PPE from leaderstats
						local maxPPE = 10 -- Default fallback
						if player:FindFirstChild("leaderstats") then
							local maxPPEValue = player.leaderstats:FindFirstChild("Max PPE")
							if maxPPEValue and maxPPEValue.Value > 0 then
								maxPPE = maxPPEValue.Value
							end
						end

						-- Get the main color based on PPE value and max PPE
						local mainColor = getPPEColor(ppeValue or 0, maxPPE)
						ppeLabel.TextColor3 = mainColor

						-- Update UIStroke color to be a darker shade
						local uiStroke = ppeLabel:FindFirstChild("UIStroke")
						if uiStroke then
							-- Create darker version by reducing the value (brightness) in HSV
							local hue, sat, val = mainColor:ToHSV()
							local darkerColor = Color3.fromHSV(hue, sat, val * 0.5) -- Make it 50% darker
							uiStroke.Color = darkerColor
						end

						-- Update PPE trend and handle transparency
						local trendInfo = updatePPETrend(player, ppeValue or 0)

						-- Handle transparency based on trend
						if trendInfo.isDecreasing and trendInfo.decreaseDuration > 25 then
							tweenFrameTransparency(innerFrame, 1, player.UserId, 10)
						elseif trendInfo.isDecreasing and trendInfo.decreaseDuration > 10 then
							-- Been decreasing for more than 5 seconds, make semi-transparent
							tweenFrameTransparency(innerFrame, 0.5, player.UserId, 0.5)
						
						
						elseif trendInfo.isIncreasing then
							-- Currently increasing, make fully opaque
							tweenFrameTransparency(innerFrame, 0, player.UserId, 0.5)
						end
					end

				end
			end
		end
	end
end

local function handleHapticFeedback(whatMotorShouldBeMoving, howLongShouldTheRumbleLast, howIntenseShouldTheRumbleBe)
	if isHapticFeedbackEnabled then
		HapticService:SetMotor(Enum.UserInputType.Gamepad1, whatMotorShouldBeMoving, howIntenseShouldTheRumbleBe)

local function handleRampHapticFeedback(whatMotorShouldBeMoving, minIntensity, maxIntensity, seconds)
	if isHapticFeedbackEnabled then
		local startTime = tick()
		local connection

		connection = RunService.Heartbeat:Connect(function()
			local elapsed = tick() - startTime
			local progress = elapsed / seconds

			if progress >= 1 then
				-- Finished ramping, set to 0
				HapticService:SetMotor(Enum.UserInputType.Gamepad1, whatMotorShouldBeMoving, 0)
				connection:Disconnect()
			else
				-- Calculate current intensity based on progress
				local currentIntensity = minIntensity + (maxIntensity - minIntensity) * progress
				HapticService:SetMotor(Enum.UserInputType.Gamepad1, whatMotorShouldBeMoving, currentIntensity)
			end
		end)
	end
end

local module = {}


local manualYchange = -2
local MinDistance = 10
local MinSeparationDistance = 6

local function BouncePetToNewPosition(player, humanoid, pet, duration)
	if pet and humanoid then
		local LookDampeningFactor = 0.1
		local YDampeningFactor = 0.02
		local followingSpeed = 0.5

		local startTime = tick()
		local startCFrame = pet.PrimaryPart.CFrame
		local PetPos = pet.PrimaryPart.Position
		local PlayerPos = humanoid.Position
		local currentY = PetPos.Y
		local folder = game.Workspace:FindFirstChild("ActivePetsClient")
		
		local resume = true
		while tick() - startTime < duration and resume do
			local distance = (pet.PrimaryPart.Position - humanoid.Position).Magnitude

			if distance < MinDistance then
				resume = false
			end
			for _, obj in pairs(folder:GetChildren()) do
				if obj:FindFirstChildWhichIsA("ObjectValue").Value == player then
					if obj:FindFirstChildWhichIsA("BoolValue").Value == false then
						resume = false
					end
				end
			end
			
			if resume then
				local t = (tick() - startTime) / duration
				local ActivePlayerPosition = humanoid.Position
				local ActivePetPosition = pet.PrimaryPart.Position

				-- Add a bouncing effect to Y-position
				local bounceHeight = math.sin(t * math.pi) * 3  -- Adjust for the desired bounce height
				local targetY = ActivePlayerPosition.Y + manualYchange

				-- Apply exponential smoothing to the Y position
				currentY = currentY * (1 - YDampeningFactor) + targetY * YDampeningFactor

				local direction = (ActivePlayerPosition - PetPos).unit

				-- Lerp for X and Z positions only
				local NewPosX = PetPos.X + (PlayerPos.X - PetPos.X) * t * followingSpeed
				local NewPosZ = PetPos.Z + (PlayerPos.Z - PetPos.Z) * t * followingSpeed

				local NewPos = Vector3.new(NewPosX, currentY + bounceHeight, NewPosZ)

				-- Add logic to move away from other pets in the same folder
				for _, otherPet in pairs(folder:GetChildren()) do
					if otherPet:IsA("Model") and otherPet ~= pet and otherPet:FindFirstChildWhichIsA("BoolValue").Value == true then
						local otherPetPos = otherPet.PrimaryPart.Position
						local distanceToOtherPetHorizontal = (Vector2.new(NewPos.X, NewPos.Z) - Vector2.new(otherPetPos.X, otherPetPos.Z)).magnitude

						if distanceToOtherPetHorizontal < MinSeparationDistance then
							local separationDirectionHorizontal = (Vector2.new(NewPos.X, NewPos.Z) - Vector2.new(otherPetPos.X, otherPetPos.Z)).unit
							local TempPos = NewPos + Vector3.new(separationDirectionHorizontal.X, 0, separationDirectionHorizontal.Y) * (MinSeparationDistance - distanceToOtherPetHorizontal)
							NewPos = Vector3.new(TempPos.X, NewPos.Y, TempPos.Z)
						end
					end
				end
				
				NewPos = PetPos:lerp(Vector3.new(NewPos.X, currentY + bounceHeight, NewPos.Z), 0.95)
				
				local playerPosition = humanoid.Position
				local direction = (playerPosition - NewPos).unit
				local currentLookVector = pet.PrimaryPart.CFrame.lookVector
				local newLookVector = currentLookVector:lerp(direction, LookDampeningFactor)
				local newCFrame = CFrame.new(NewPos, NewPos + newLookVector)

				pet:SetPrimaryPartCFrame(newCFrame)
				task.wait()
			end			
		end
	end
end


local function movePetWhenCloseToPlayer(humanoid, pet)
	local dampeningFactor = 0.025
	local fixedDistance = 8
	local directionToPlayer = (humanoid.Position - pet.PrimaryPart.Position).unit
	local NewPos = humanoid.Position - directionToPlayer * fixedDistance
	local currentCFrame = pet.PrimaryPart.CFrame
	
	
	local folder = game.Workspace:FindFirstChild("ActivePetsClient")
	for _, otherPet in pairs(folder:GetChildren()) do
		if otherPet:IsA("Model") and otherPet ~= pet then
			local otherPetPos = otherPet.PrimaryPart.Position
			local distanceToOtherPetHorizontal = (Vector2.new(NewPos.X, NewPos.Z) - Vector2.new(otherPetPos.X, otherPetPos.Z)).magnitude

			if distanceToOtherPetHorizontal < MinSeparationDistance then
				local separationDirectionHorizontal = (Vector2.new(NewPos.X, NewPos.Z) - Vector2.new(otherPetPos.X, otherPetPos.Z)).unit
				local TempPos = NewPos + Vector3.new(separationDirectionHorizontal.X, 0, separationDirectionHorizontal.Y) * (MinSeparationDistance - distanceToOtherPetHorizontal)
				NewPos = Vector3.new(TempPos.X, NewPos.Y, TempPos.Z)
			end
		end
	end
	
	local newCFrame = currentCFrame:Lerp(CFrame.new(NewPos,	NewPos + directionToPlayer), dampeningFactor)

	local PetPos = pet.PrimaryPart.CFrame.Position
	local PlayerPos = humanoid.Position
	local currentY = pet.PrimaryPart.CFrame.Y
	local targetY = PlayerPos.Y
	currentY = currentY * (1 - dampeningFactor) + targetY * dampeningFactor

	local finalVector3= Vector3.new(newCFrame.X, currentY, newCFrame.Z)

	local finalCFrame = CFrame.new(
		finalVector3,
		finalVector3 + newCFrame.LookVector -- Dampened look vector
	)

	pet:SetPrimaryPartCFrame(finalCFrame)
end

function TeleportModule.Init()
	changeSkySettings(game.Players.LocalPlayer, "1")
	-- Set initial level
	currentLevel = "1"
	local teleporting = false

	for _, level in pairs(rootMapFolder:GetChildren()) do
		local levelName = level.Name:sub(1, 1)

		local function combTable(tables)
			local final = {}
			for _, firstlvltable in pairs(tables) do
				for _, secondlvltable in pairs(firstlvltable) do
					table.insert(final,secondlvltable)
				end
			end
			return(final)
		end

		-- Make ALL teleport parts transparent (destination parts, forward parts, and back parts)
		for _, part in pairs(combTable({
			TeleportModule.ReturnTeleportParts(levelName, "EndDestination"),
			TeleportModule.ReturnTeleportParts(levelName, "StartDestination"),
			TeleportModule.ReturnTeleportParts(levelName, "Forwards"),
			TeleportModule.ReturnTeleportParts(levelName, "Back")})) do
			if part:IsA("Part") then
				part.CanCollide = false
				part.Transparency = 1
			end
		end

		local OriginParts = TeleportModule.ReturnTeleportParts(levelName, "Forwards")
		local Destination = nil
		for _, object in pairs(OriginParts) do
			if object:IsA("ObjectValue") then
				Destination = object.Value
			end
		end

		local DestinationParts = nil
		local DestinationLvlName = nil
		if Destination then
			DestinationLvlName = Destination.Name:sub(1, 1)
			DestinationParts = TeleportModule.ReturnTeleportParts(DestinationLvlName, "StartDestination")
		end

		if DestinationParts then
			for _, part in pairs(OriginParts) do
				if part:IsA("Part") then
					part.Touched:Connect(function(hit)
						local humanoid = hit.Parent:FindFirstChild("Humanoid")
						if humanoid and game.Players:GetPlayerFromCharacter(hit.Parent) and not teleporting then
							teleporting = true
							local player = game.Players:GetPlayerFromCharacter(hit.Parent)
							local randomInt = math.random(1, #DestinationParts)
							TeleportModule.TeleportPlayerToRandomPlaceInLvl(player, DestinationLvlName)
							wait(1)
							teleporting = false
						end
					end)
				end
			end
		end

		--backwards part
		local levelName = level.Name:sub(1, 1)
		local OriginParts = TeleportModule.ReturnTeleportParts(levelName, "Back")
		local Destination = nil
		for _, object in pairs(OriginParts) do
			if object:IsA("ObjectValue") then
				Destination = object.Value
			end
		end

		local DestinationParts = nil
		local DestinationLvlName = nil
		if Destination then
			DestinationLvlName = Destination.Name:sub(1, 1)
			DestinationParts = TeleportModule.ReturnTeleportParts(DestinationLvlName, "EndDestination")
		end

		if DestinationParts then
			for _, part in pairs(OriginParts) do
				if part:IsA("Part") then

						local humanoid = hit.Parent:FindFirstChild("Humanoid")

-- MapOptimization ModuleScript
-- Manages hiding/showing objects across multiple levels for performance optimization

local MapOptimization = {}

-- Public table to store all hidden objects - accessible by other scripts
MapOptimization.HiddenObjects = {}

-- Private table to store objects by level for efficient level-based operations
MapOptimization.objectsByLevel = {}

-- Services
local Workspace = game:GetService("Workspace")

-- Initialize the module by scanning all levels for objects marked for optimization
function MapOptimization.init()
	local levelsFolder = Workspace:FindFirstChild("Levels")
	if not levelsFolder then
		warn("MapOptimization: Levels folder not found in Workspace")
		return
	end

	-- Clear existing data
	MapOptimization.HiddenObjects = {}
	MapOptimization.objectsByLevel = {}

	--print("MapOptimization: Found Levels folder with " .. #levelsFolder:GetChildren() .. " children")

	-- Scan all levels
	for _, levelFolder in pairs(levelsFolder:GetChildren()) do
		--print("MapOptimization: Checking child: " .. levelFolder.Name .. " (Type: " .. levelFolder.ClassName .. ")")

		if levelFolder:IsA("Folder") then
			local levelName = levelFolder.Name -- Just use the folder name directly
			--print("MapOptimization: Processing level folder: '" .. levelName .. "'")

			MapOptimization.objectsByLevel[levelName] = {}
			MapOptimization:ScanLevelForOptimizationObjects(levelFolder, levelName)
			--print("MapOptimization: Scanned level '" .. levelName .. "' - Found " .. #MapOptimization.objectsByLevel[levelName] .. " objects")
		end
	end

	-- Debug: Print all level names that were found
	--print("MapOptimization: Levels registered:")
	for levelName, objects in pairs(MapOptimization.objectsByLevel) do
		--print("  - '" .. levelName .. "' with " .. #objects .. " objects")
	end

	-- Automatically unload all levels (hide all optimization objects)
	local totalUnloaded = 0
	--print("MapOptimization: Auto-unloading all levels...")

	for levelName, objects in pairs(MapOptimization.objectsByLevel) do
		local unloadedCount = 0
		for _, obj in pairs(objects) do
			if obj and obj.Parent then
				-- Store original parent information before hiding
				obj:SetAttribute("OriginalParentPath", obj.Parent.Name)
				obj.Parent = nil
				unloadedCount = unloadedCount + 1
			end
		end
		totalUnloaded = totalUnloaded + unloadedCount
		--print("MapOptimization: Auto-unloaded level '" .. levelName .. "' - Hidden " .. unloadedCount .. " objects")
	end

	--print("MapOptimization: Initialization complete. Total objects tracked: " .. #MapOptimization.HiddenObjects .. ", Auto-unloaded: " .. totalUnloaded)
end

-- Get the level name from a level folder (just use the folder name)
function MapOptimization:GetLevelName(levelFolder)
	return levelFolder.Name
end

-- Find a level folder by its name (just match folder names)
function MapOptimization:FindLevel(levelName)
	local levelsFolder = Workspace:FindFirstChild("Levels")
	if not levelsFolder then
		return nil
	end

	return levelsFolder:FindFirstChild(levelName)
end

-- Recursively scan a level for objects marked with "RemoveForOptimisation"
function MapOptimization:ScanLevelForOptimizationObjects(parent, levelName)
	for _, child in pairs(parent:GetChildren()) do
		-- Check if this object has the optimization marker
		if child:FindFirstChild("RemoveForOptimisation") then
			-- Add to both tracking tables
			table.insert(MapOptimization.HiddenObjects, child)
			table.insert(MapOptimization.objectsByLevel[levelName], child)
		end

		-- Recursively scan children
		MapOptimization:ScanLevelForOptimizationObjects(child, levelName)
	end
end

-- Hide/unload a specific level
function MapOptimization:UnloadLevel(levelName)
	if not MapOptimization.objectsByLevel[levelName] then
		warn("MapOptimization: Level '" .. levelName .. "' not found or not initialized")
		return false
	end

	local hiddenCount = 0

	for _, obj in pairs(MapOptimization.objectsByLevel[levelName]) do
		if obj and obj.Parent then
			obj.Parent = nil
			hiddenCount = hiddenCount + 1
		end
	end

	--print("MapOptimization: Unloaded level '" .. levelName .. "' - Hidden " .. hiddenCount .. " objects")
	return true
end

-- Show/load a specific level
function MapOptimization:LoadLevel(levelName)
	local levelFolder = MapOptimization:FindLevel(levelName)
	if not levelFolder then
		warn("MapOptimization: Level folder for '" .. levelName .. "' not found")
		return false
	end

	if not MapOptimization.objectsByLevel[levelName] then
		warn("MapOptimization: Level '" .. levelName .. "' not found in optimization data")
		return false
	end

	local restoredCount = 0

	for _, obj in pairs(MapOptimization.objectsByLevel[levelName]) do
		if obj and not obj.Parent then
			-- Find the original parent by checking if the object's expected parent exists in the level
			local success = MapOptimization:RestoreObjectToOriginalParent(obj, levelFolder)
			if success then
				restoredCount = restoredCount + 1
			end
		end
	end

	--print("MapOptimization: Loaded level '" .. levelName .. "' - Restored " .. restoredCount .. " objects")
	return true
end

-- Attempt to restore an object to its original parent within the level
function MapOptimization:RestoreObjectToOriginalParent(obj, levelFolder)
	-- Store original parent reference when first hiding (this is a simplified approach)
	-- In a more complex system, you might want to store the full path
	if not obj:GetAttribute("OriginalParentPath") then
		-- If no stored path, try to place it back in the level folder as fallback
		obj.Parent = levelFolder
		return true
	end

	-- Try to restore to original location (this would need more complex path resolution)
	-- For now, just restore to level folder
	obj.Parent = levelFolder
	return true
end

-- Hide a specific object (and store its original parent path)
function MapOptimization:HideObject(obj)
	if obj and obj.Parent then
		-- Store original parent information
		obj:SetAttribute("OriginalParentPath", obj.Parent.Name)
		obj.Parent = nil

		-- Add to hidden objects if not already tracked
		if not table.find(MapOptimization.HiddenObjects, obj) then
			table.insert(MapOptimization.HiddenObjects, obj)
		end

		return true
	end
	return false
end

-- Show a specific object
function MapOptimization:ShowObject(obj, targetParent)
	if obj and not obj.Parent and targetParent then
		obj.Parent = targetParent
		return true
	end
	return false
end

-- Get all objects for a specific level
function MapOptimization:GetLevelObjects(levelName)
	return MapOptimization.objectsByLevel[levelName] or {}
end

-- NEW FUNCTION: Get all objects from a level (both hidden and visible)
function MapOptimization:GetAllLevelObjects(levelName)
	local allParts = {}

	-- Add hidden/tracked objects
	if MapOptimization.objectsByLevel[levelName] then
		for _, obj in pairs(MapOptimization.objectsByLevel[levelName]) do
			if obj then
				table.insert(allParts, obj)
			end
		end
	end

	-- Add visible parts
	local levelFolder = MapOptimization:FindLevel(levelName)
	if levelFolder then
		MapOptimization:RecursivelyFindParts(levelFolder, allParts)
	end

	-- Remove duplicates
	local uniqueParts = {}
	local seen = {}

	for _, part in pairs(allParts) do
		if not seen[part] then
			seen[part] = true
			table.insert(uniqueParts, part)
		end
	end

	-- Create folder of clones
	local partsFolder = Instance.new("Folder")
	partsFolder.Name = levelName .. "_Parts"

	for _, part in ipairs(uniqueParts) do
		local clone = part:Clone()
		clone.Parent = partsFolder
	end

	return partsFolder
end


-- Helper function to recursively find all BaseParts in a container
function MapOptimization:RecursivelyFindParts(parent, partsTable)
	for _, child in pairs(parent:GetChildren()) do
		if child:IsA("BasePart") then
			table.insert(partsTable, child)
		end

		-- Recursively search children (but skip objects that have RemoveForOptimisation 
		-- marker to avoid counting them twice if they're currently visible)
		if not child:FindFirstChild("RemoveForOptimisation") then
			MapOptimization:RecursivelyFindParts(child, partsTable)
		end
	end
end

-- Check if a level is currently loaded (has objects with parents)
function MapOptimization:IsLevelLoaded(levelName)
	if not MapOptimization.objectsByLevel[levelName] then
		return false
	end

	for _, obj in pairs(MapOptimization.objectsByLevel[levelName]) do
		if obj and obj.Parent then
			return true
		end
	end

	return false
end

-- Get statistics about the optimization system
function MapOptimization:GetStats()
	local stats = {
		totalTrackedObjects = #MapOptimization.HiddenObjects,
		levelsTracked = 0,
		currentlyHiddenObjects = 0
	}

	for levelName, objects in pairs(MapOptimization.objectsByLevel) do
		stats.levelsTracked = stats.levelsTracked + 1
		for _, obj in pairs(objects) do
			if obj and not obj.Parent then
				stats.currentlyHiddenObjects = stats.currentlyHiddenObjects + 1
			end
		end
	end

	return stats
end

return MapOptimization
